// fonction qui additionne 2 images
function render=addition(img1,img2)
    //Largeur et hauteur
    [wd,he]=size(img1);
    // On créer une image vierge pour la remplir de la soustraction
    render = zeros(wd,he);
    // Pour chaque colonne
    for i=1:he
        // Pour chaque ligne
        for j=1:wd
            pix1 = img1(j,i);
            pix2 = img2(j,i);

// on rentre la soustraction dans l'image vierge
            render(j,i) = max(pix1 - pix2,0);
        end
    end
endfunction
